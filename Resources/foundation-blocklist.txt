# Symbols intentionally excluded from auto-harvest because the
# generator's bridge would diverge from a hand-rolled implementation
# elsewhere in the interpreter. Each entry is the same path-component
# format as the allowlist.

# The single-arg primitive constructors are hand-rolled in
# `Sources/SwiftScriptInterpreter/Builtins/IOBuiltins.swift` to handle
# the multi-shape coercion (`Int("42") -> Int?`, `Int(3.14) -> Int`,
# etc.). The auto-bridge would only model one shape.
Int.init(_:)
Double.init(_:)
Bool.init(_:)
String.init(_:)

# `print(_:)` — hand-rolled to support `print(a, b, c)` variadic-style
# calls and `print(_:separator:terminator:)` with labels. The auto-
# bridge would only model the single-arg form.
print(_:)

# `readLine(strippingNewline:)` — hand-rolled in `IOBuiltins`. Auto
# bridges as `Foundation.readLine` which doesn't exist.
readLine()
readLine(strippingNewline:)

# `String.hasPrefix`/`hasSuffix`/`contains` — hand-rolled in
# `Interpreter+Members.swift` (the inline `.string` switch). The
# auto-bridge collides at register-time but the inline switch fires
# first; cleaner to skip the bridge so it doesn't sit as dead code.
String.hasPrefix(_:)
String.hasSuffix(_:)

# `String.uppercased()` and `lowercased()` — hand-rolled inline.
String.uppercased()
String.lowercased()

# `Int.isMultiple(of:)` — auto-harvest emits an `extensions["Int"]`
# method, but we want `5.isMultiple(of:)` to use the inline switch path.
# Blocking is harmless because the bridge wouldn't be reached.
Int.isMultiple(of:)

# `Double.squareRoot()` / `Double.rounded()` — inline-cased.
Double.squareRoot()
Double.rounded()

# `Double.rounded(_:)` — we hand-rolled the rule-arg form to support
# implicit-member context (`d.rounded(.up)`).
Double.rounded(_:)

# `String.distance(from:to:)` and similar require String.Index, which
# we don't model. They'd compile (since their signature is bridgeable
# in shape) but always fail at runtime with malformed-Index errors.
StringProtocol.contains(_:)

# `pow(Decimal, Int) -> Decimal` shadows the C-math `pow(Double, Double)`
# we hand-roll for math scripts. The Decimal overload is rarely what
# scripts want.
pow(_:_:)

# macOS-only symbols that the macOS-extracted symbol graph reports as
# universally available but the iOS Foundation overlay doesn't expose.
# Blocking here keeps the package buildable on both platforms.
ProgressUserInfoKey.fileAnimationImageKey
ProgressUserInfoKey.fileAnimationImageOriginalRectKey
ProgressUserInfoKey.fileIconKey
URLResourceKey.applicationIsScriptableKey
URLResourceKey.tagNamesKey
URLResourceKey.quarantinePropertiesKey
NSHFSTypeOfFile(_:)

# `FileManager.default` and its instance methods stay hand-rolled in
# Interpreter+FileIO.swift — they dispatch through a `.structValue`
# sentinel rather than the standard `.opaque` receiver flow, and the
# auto-bridge skips a couple of the actually-used methods (e.g.
# `contentsOfDirectory(atPath:)`). Block here so the hand-rolled
# sentinel wins the lookup.
FileManager.default

# Among the URLSession async overloads, prefer the simpler URL-based
# variants. Blocking the URLRequest-based ones lets the (from: URL)
# overload win the deduper.
URLSession.data(for:)
URLSession.data(for:delegate:)
URLSession.data(from:delegate:)
URLSession.bytes(for:)
URLSession.bytes(for:delegate:)
URLSession.bytes(from:delegate:)
URLSession.download(for:)
URLSession.download(for:delegate:)
URLSession.download(from:delegate:)
URLSession.download(resumeFrom:)
URLSession.download(resumeFrom:delegate:)
URLSession.upload(for:from:)
URLSession.upload(for:from:delegate:)
URLSession.upload(for:fromFile:delegate:)
